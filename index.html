###index

- git init

- git add README.md
- git commit -m "feat: add README-md"
- git flow feature finish 'feature-name

브랜치에서 push 하지않고 merge(finish)하게 되면 feature가 사라짐
push 하고 finish 하게 되면 기록이 다 남음(feature)가 남음.

- git flow release start v0.0.1.00190320001
- git flow release finish v0.0.1.00190320001


v0.0.1.00190320001 = 버전 이름 붙이기 00 = 2000년대 
                        19 = 2019년 
                        03=3월 
                        20=20일 
                        001=버전 1
v0.0.1.00190320001a = 알파버전


- 협업하기 - github fork 누르고 이동하는 곳에서 클론 이후에 master에서 git branch develop 만들기 -> 
            git pull origin devleop 해서 작업 시작 (master에서 끌어옴)

작업 이후 devlop으로 push. 그 이후 github에서 pull-request 버튼 클릭 그 이후 base:develop으로 설정. master로 설정하면 바로 검증없이 live로 넘어가버림

    실시간으로 변하는 원본 레포를 반영하기 위해 한 번 pull request 한 뒤 git remote add rmorigin –주소– 하면 새롭게 바뀌는 pm의 레포를 떠올 수 있다.

    <input type="text" name="content1" id="content1">
    <input type="text" name="content2" id="content2">
    <input type="text" name="content3" id="content3">
    <input type="text" name="content" id="content">
    <button id="btn_save">Save</button>
    <button id="btn_load">Load</button>
</form>


<script type="text/javascript">

        // 변수로 storage_key의 값을 선언 (form의 id)
        // local storage로 저장을 위한 식
        var storage_key = "content";
        $(function() {
            $('#btn_save').click(function() {
                // 폼 데이터를 문자열로 합치기
                var form_content = $('#content_form').serialize();

                // 합친 문자열을 localStorage에 저장
                localStorage.setItem(storage_key, form_content);

                //폼 submit이 일어나지 않게
                return false;
            });

        // local storage에 저장된 정보를 다시 불러오는 식
        $('#btn_load').click(function() {
            form_content = localStorage.getItem(storage_key);

            // 각 정보들 사이가 &로 연결되어 저장되는데 이를 split을 이용해서 분리, 각 항에 다시 표시한다.
            var split_data = form_content.split("&");
            for ( var i in split_data) {
                var content_data = split_data[i].split("=");
                var key = content_data[0];
                var data = content_data[1];
                $('#'+key).val(data);
            }
            return false;
</script>

Validation check

오늘 수업 복습 (3월 18일 월요일)

회원가입 창 형태를 만들고 ‘제출하기 버튼’ 클릭 시 각 input 창에 입력된 정보값의 validation check하기

submit을 사용하기 위해서 input과 button들을 form에 집어넣고 시작

<script type="text/javascript">
        // 폼을 사용함으로서 submit 명령을 사용가능하게 함.
        $("#form").submit(function(e) {

        // 이메일과 비밀번호의 형식을 정하고 맞추기 위한 정규표현식
            var emailRule = /^[0-9a-zA-Z]([-_.]?[0-9a-zA-Z])*@[0-9a-zA-Z]( [-_.]?[0-9a-zA-Z])*.[a-zA-Z]{2,3}$/i;
            var passRule = /^[A-Za-z0-9]{6,12}$/;

        //정규표현식을 통해 잘못된 형식을 걸러내기 전 input란이 비어있을 경우  값 입력을 요구하는 경고
            if($("#identification").val()=='') {
                alert("이메일을 입력해주세요");
                return false;
            }
        //정규표현식을 통해서 틀린 형식의 이메일을 걸러내고 재입력을 요구하는 경고창
            if(!emailRule.test($("#identification").val())) {
                alert("이메일 형식이 틀렸습니다");
                return false;
            }
        //6~12자리의 패스워드인지 판별하기 전, 빈 값을 채워달라는 경고창
            if($("#password").val=='') {
                alert("패스워드를 입력해주세요");
                return false;
            }
        //패스워드가 6~12자리인지 검사하고 형식이 틀렸을 경우 다시 돌려보내는 경고창
            if(!passRule.test($("#password").val())) {
                alert("패스워드 형식이 틀렸습니다")
                return false;
            }
            if($('#name').val()=='') {
                alert('이름을 입력하세요');
                return false;
            }
            if($('#address').val()=='') {
                alert('주소를 입력하세요');
                return false;
            }

        //e.preventDefault를 통해 전체에 입력된 값이 지워지지 않고 확인을 누르고 난 뒤에도 남아있게 만듦 + 위의 모든 조건을 통과하면 회원가입완료 창이 뜨게 함.
            e.preventDefault();
            alert("회원가입완료");
})
</script>

How to make random number generator.

랜덤 번호 생성하기

JavaScript를 이용해 일정범위의 숫자들을 무작위로 뽑아내는 랜덤 번호 생성기 만들기를 배웠다.
아직은 JavaScript를 이용하는데에 제대로 이해가 되지않아 여러번 코드를 다시봐야하지만,
그래도 하고싶었던 일에 도전한다는 것이 즐겁고 재미있다고 생각한다. 사실 즐겁고 재밌다고
생각한다는 것이 아직 내 노력이 부족한것이 아닐까라는 생각이 들어 좀 더 치열하게 공부해야겠다는
생각이 든다.

오늘 배운 코드 정리

<script type="text/javascript">
    // 같은 코드를 총 5번 반복하기 위해 for 반복문을 이용
        for(j=0;j<5;j++) {

    // var 을 이용해 변수를 선언하고 original_numbers의 공간을 빈 공간으로 설정
        var original_numbers = [];

    // 1부터 45사이의 숫자 범위를 정하기 위해 for 반복문을 이용해서 작성
    // for 반복분은 초기식이 실행된 후에 다시 한번 조건을 비교할 때 사용
        for(i=1;i<45;i++) {

    // push 명령을 통해 i 값을 출력
            original_numbers.push(i);
        }


    var numbers = []; //숫자를 담을 공간

    // for문을 이용한 반복으로 6개의 랜덤한 숫자를 뽑아내고 6개의 랜덤한 숫자가 뽑혔을 경우 종료
    for(i=0;i<6;i++) {
        var number_index = Math.floor((Math.random() * original_numbers.length));
        var get_number = original_numbers[number_index];
        numbers.push(get_number);
        original_numbers.splice(number_index,1)
    }

    //뽑힌 6개의 숫자들을 작은 숫자부터 큰 숫자 순서대로 나열하기
    numbers.sort(function(a, b) {return a-b});
    console.log(numbers);
    }
</script>

what-have-I-done-to-you-node?

처음 다뤄본 nodejs가 빚은 오늘의 멘붕

처음 컴퓨터에 설치한 nodejs는 혼돈 그 자체였다.
아무것도 모른채 홈페이지에 들어가 다운로드를 받았고,
그 때까지 나는 내 컴퓨터가 mac이라는 것을 간과하고 있었다.

아무것도 모른채 pkg 파일을 설치하고 터미널에 들어가
다시 homebrew를 통해 설치하려고 하니 중간중간 나타나는 error
다행히 몇몇 기능에서 오류를 보이긴하지만 수업 중 과제를 하는데에는
별 다른 문제가 없어수업은 아무 문제 없이 진행을 할 수 있었다.
하지만 향후에 문제가 생길 것이 걱정된 나는 이미 깔려있는
node를 완벽하게 삭제하고 다시 설치할 생각을 하게 된다.

google을 통해 node 완벽 삭제 방법을 찾았고 그 방법대로
실행에 옮겼지만 결과는… node 연결 폴더들만 삭제되고 터미널 내에서는
node라는 존재만 인식되는 상황이 되어버렸다.

다시 google을 돌아다니면서 방법을 찾아다녔지만 node의 특성상
다양한 디렉토리에 자신을 깔아놓고 있었고, 완벽한 삭제는 거의
불가능한 상황처럼 보였다.

강사님 함께 컴퓨터를 싸악 밀고 다시 설치해야하 하는지에 대한
고민을 하며 마지막으로 시도한 방식은 nvm을 통한 설치와 관리였다.

다행히도 nvm은 정상적으로 설치하고 작동해주었고 덕분에 가장
node 관리에 좋다는 nvm을 반 강제적으로 사용할 수 있게 되었다.
