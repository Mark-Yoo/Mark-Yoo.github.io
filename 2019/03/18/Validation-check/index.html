Validation check

오늘 수업 복습 (3월 18일 월요일)

회원가입 창 형태를 만들고 ‘제출하기 버튼’ 클릭 시 각 input 창에 입력된 정보값의 validation check하기

submit을 사용하기 위해서 input과 button들을 form에 집어넣고 시작

<script type="text/javascript">
        // 폼을 사용함으로서 submit 명령을 사용가능하게 함.
        $("#form").submit(function(e) {

        // 이메일과 비밀번호의 형식을 정하고 맞추기 위한 정규표현식
            var emailRule = /^[0-9a-zA-Z]([-_.]?[0-9a-zA-Z])*@[0-9a-zA-Z]( [-_.]?[0-9a-zA-Z])*.[a-zA-Z]{2,3}$/i;
            var passRule = /^[A-Za-z0-9]{6,12}$/;

        //정규표현식을 통해 잘못된 형식을 걸러내기 전 input란이 비어있을 경우  값 입력을 요구하는 경고
            if($("#identification").val()=='') {
                alert("이메일을 입력해주세요");
                return false;
            }
        //정규표현식을 통해서 틀린 형식의 이메일을 걸러내고 재입력을 요구하는 경고창
            if(!emailRule.test($("#identification").val())) {
                alert("이메일 형식이 틀렸습니다");
                return false;
            }
        //6~12자리의 패스워드인지 판별하기 전, 빈 값을 채워달라는 경고창
            if($("#password").val=='') {
                alert("패스워드를 입력해주세요");
                return false;
            }
        //패스워드가 6~12자리인지 검사하고 형식이 틀렸을 경우 다시 돌려보내는 경고창
            if(!passRule.test($("#password").val())) {
                alert("패스워드 형식이 틀렸습니다")
                return false;
            }
            if($('#name').val()=='') {
                alert('이름을 입력하세요');
                return false;
            }
            if($('#address').val()=='') {
                alert('주소를 입력하세요');
                return false;
            }

        //e.preventDefault를 통해 전체에 입력된 값이 지워지지 않고 확인을 누르고 난 뒤에도 남아있게 만듦 + 위의 모든 조건을 통과하면 회원가입완료 창이 뜨게 함.
            e.preventDefault();
            alert("회원가입완료");
})
</script>
