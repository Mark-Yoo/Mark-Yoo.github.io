How to make random number generator.

랜덤 번호 생성하기

JavaScript를 이용해 일정범위의 숫자들을 무작위로 뽑아내는 랜덤 번호 생성기 만들기를 배웠다.
아직은 JavaScript를 이용하는데에 제대로 이해가 되지않아 여러번 코드를 다시봐야하지만,
그래도 하고싶었던 일에 도전한다는 것이 즐겁고 재미있다고 생각한다. 사실 즐겁고 재밌다고
생각한다는 것이 아직 내 노력이 부족한것이 아닐까라는 생각이 들어 좀 더 치열하게 공부해야겠다는
생각이 든다.

오늘 배운 코드 정리

<script type="text/javascript">
    // 같은 코드를 총 5번 반복하기 위해 for 반복문을 이용
        for(j=0;j<5;j++) {

    // var 을 이용해 변수를 선언하고 original_numbers의 공간을 빈 공간으로 설정
        var original_numbers = [];

    // 1부터 45사이의 숫자 범위를 정하기 위해 for 반복문을 이용해서 작성
    // for 반복분은 초기식이 실행된 후에 다시 한번 조건을 비교할 때 사용
        for(i=1;i<45;i++) {

    // push 명령을 통해 i 값을 출력
            original_numbers.push(i);
        }


    var numbers = []; //숫자를 담을 공간

    // for문을 이용한 반복으로 6개의 랜덤한 숫자를 뽑아내고 6개의 랜덤한 숫자가 뽑혔을 경우 종료
    for(i=0;i<6;i++) {
        var number_index = Math.floor((Math.random() * original_numbers.length));
        var get_number = original_numbers[number_index];
        numbers.push(get_number);
        original_numbers.splice(number_index,1)
    }

    //뽑힌 6개의 숫자들을 작은 숫자부터 큰 숫자 순서대로 나열하기
    numbers.sort(function(a, b) {return a-b});
    console.log(numbers);
    }
</script>
